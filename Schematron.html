def self.create_from_file(sf)
  if sf.kind_of? IO
    sf = SchematronFile.new(sf.read)
  end

  me = Schematron.find_by(digest: sf.digest) || Schematron.new(digest: sf.digest,
                                                               issues_attributes: sf.issue_attrs)
  if me.persisted?
    me.touch
  else
    me.save
  end

  me
end
