def faid_attr
  xml = Nokogiri::XML(self)
  xml.remove_namespaces!
  eadid = xml.xpath('/ead/eadheader/eadid')[0]
  {
    eadid: eadid.text,
    ext_id_type: 'hollis',
    ext_id: eadid['identifier'],
    repository_id: Repository.find_by(code: eadid.text[0..2]).id
  }
end
