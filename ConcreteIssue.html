def generate_tags_from_diagnostic_info
  self.tags = diagnostic_info
              .split("\n")
              .map {|s| s.match(/([^\s:]+): (.+)/)}
              .reject(&:blank?)
              .map {|m| m[1..2]}
              .to_h
end
