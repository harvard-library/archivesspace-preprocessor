def self.create_from_file(fa)
  if fa.kind_of? IO
    fa = FindingAidFile.new(fa.read)
  end

  attr = fa.fav_attr
  FindingAidVersion.find_by(digest: attr[:digest]) || FindingAidVersion.create(attr)
end
